# Introduction

## install

```bash
yarn add @leafage-libs/lp-utils
```

## usage

```js
import '@leafage-gk/lp-utils/dist/lp-utils.css'
import Vue from 'vue'
import VueHead from 'vue-head'
import LpUtils, { createPage } from '@leafage-gk/lp-utils'

Vue.use(VueHead)
Vue.use(LpUtils)

createPage({
  head: {
    title: {
      inner: 'page',
    },
  },
  sections: [],
})
```
